数组中的第K个最大元素#

链接: https://leetcode.cn/problems/kth-largest-element-in-an-array/

1、题目描述#

2、思路说明#

利用快排的思想解决该问题。

3、代码#

class Solution:

    def reverse(self, nums, start, end, k):
        # 边界条件
        if start > end:
            return None
        if start == end:
            if len(nums) - start == k:
                return start
            else:
                return None


        cache = nums[start]
        l, r = start, end

        while l < r:
            # 此时位置 l 处为空闲
            while nums[r] >= cache and l < r:
                r -= 1
            nums[l] = nums[r]

            # 此时位置 r 处为空闲
            while nums[l] <= cache and l < r:
                l += 1
            nums[r] = nums[l]

        nums[l] = cache
        # 此时必然有 l == r，且位置 l 左侧的值都比 cache 小，位置 l 右侧的值都比 cache 大

        if len(nums) - l == k:  # 如果位置 l 就是第 k 大的数
            return l
        if len(nums) - l > k:
            return self.reverse(nums, l + 1, end, k)  # 如果位置 l 的数比第 k 大的数要小，那么就在右侧找
        if len(nums) - l < k:
            return self.reverse(nums, start, l - 1, k)  # 如果位置 l 的数比第 k 大的数要大，那么就在左侧找

    def findKthLargest(self, nums: List[int], k: int) -> int:
        idx = self.reverse(nums, 0, len(nums) - 1, k)
        if idx is not None and 0 <= idx < len(nums):
            return nums[idx]
        raise RuntimeError(f"ERROR, nums length: {len(nums)}, idx: {idx}")
